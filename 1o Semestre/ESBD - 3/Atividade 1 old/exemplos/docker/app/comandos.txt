docker build --tag my_app .             # cria a imagem de nome "my_app" a partir das instruções da Dockerfile
docker run -d --name app my_app         # cria um container com o nome "app" a partir da imagem de nome "my_app" de forma detatched (sem stdout/err)

docker ps                               # mostra todos os containers
docker ps -a                            # mostra todos os containers, inclusive parados
docker images                           # mostra todas as imagens
docker stop <NOME_OU_ID_DO_CONTAINER>   # para o container
docker rm <NOME_OU_ID_DO_CONTAINER>     # remove o containers
